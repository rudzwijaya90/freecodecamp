function titleCase(str) {
  const wordArr = str.split(" ");
  const updatedTitle = [];
  for (let st in wordArr) {
    updatedTitle[st] = wordArr[st][0].toUpperCase() + wordArr[st].slice(1).toLowerCase();
  }
  return updatedTitle.join(" ");
}
